% Production Rules:
route(Town1, Town2, Distance) :- 
    road(Town1, Town2, Distance).
route(Town1, Town2, Distance) :-
    road(Town1, X, Dist1),
    route(X, Town2, Dist2),
    Distance is Dist1 + Dist2.

% Domains:
town = symbol.
distance = integer.

% Clauses:
road("tampa", "houston", 200).
road("gordon", "tampa", 300).
road("houston", "gordon", 100).
road("houston", "kansas_city", 120).
road("gordon", "kansas_city", 130).
